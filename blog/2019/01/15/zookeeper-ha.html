                        # 单机安装

                        # 下载解压

                        wget https://archive.apache.org/dist/zookeeper/zookeeper-3.4.13/zookeeper-3.4.13.tar.gz
tar -zxvf zookeeper-3.4.13.tar.gz
mv zookeeper-3.4.13 zookeeper
                        1
                        2
                        3

                        # 配置环境变量，编辑/etc/profile文件

                        export ZOOKEEPER_HOME=/data/zookeeper/zookeeper
export PATH=$PATH:$ZOOKEEPER_HOME/bin
                        1
                        2

                        执行source /etc/profile命令

                        # 编辑配置文件

                        进入conf文件夹下新增编辑zoo.cfg文件并指定dataDir和dataLogDir的值

                        cd /data/zookeeper/zookeeper
mkdir dataDir
mkdir dataLogDir
cd conf
cp zoo_sample.cfg zoo.cfg

                        # 下载解压ZK

                        wget https://archive.apache.org/dist/zookeeper/zookeeper-3.4.11/zookeeper-3.4.11.tar.gz
tar -zxvf zookeeper-3.4.11.tar.gz
                        1
                        2

                        # 配置环境变量

                        编辑/etc/profile文件，在最后添加

                        export ZOOKEEPER_HOME=/data/zookeeper/zookeeper-3.4.11
export PATH=$PATH:$ZOOKEEPER_HOME/bin
                        1
                        2

                        # 配置生效

                        source /etc/profile
                        1

                        # 编辑配置文件

                        进入conf文件下下新增编辑zoo.cfg文件

                        tickTime=2000
initLimit=10
syncLimit=5
dataDir=/data/zookeeper/zookeeper-3.4.11/dataDir
dataLogDir=/data/zookeeper/zookeeper-3.4.11/dataLogDir
clientPort=2181
4lw.commands.whitelist=*
server.1=192.168.0.55:2888:3888
server.2=192.168.0.233:2888:3888
server.3=192.168.0.195:2888:3888
