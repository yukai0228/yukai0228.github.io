                  # Elasticsearch分布式集群搭建

                  发布于 01/16/2019, 3:09 AM

                  # 下载解压ES

                  wget https://artifacts.elastic.co/downloads/elasticsearch/elasticsearch-6.4.2.tar.gz
tar -zxvf elasticsearch-6.4.2.tar.gz
                  1
                  2

                  # 创建数据与日志目录

                  [root@server-0005 elasticsearch]# mkdir data
[root@server-0005 elasticsearch]# mkdir logs
                  1
                  2

                  # 配置elasticsearch.yml文件

                  依次配置三台服务器，注意替换节点名和节点地址

                  cluster.name: es-cluster
node.name: node-1 # 节点名
path.data: /data/elasticsearch/data
path.logs: /data/elasticsearch/logs
bootstrap.memory_lock: false
bootstrap.system_call_filter: false
network.host: 192.168.0.195 # 节点地址
http.port: 9200
discovery.zen.ping.unicast.hosts: ["192.168.0.195:9300","192.168.0.5:9300","192.168.0.55:9300"]
discovery.zen.minimum_master_nodes: 2
gateway.recover_after_nodes: 3
node.master: true
node.data: true

                  # 创建用户

                  由于es不能用root用户启动，所以我们新建一个用户

                  [root@server-0005 bin]# groupadd esgroup
[root@server-0005 bin]# useradd esroot -g esgroup -p espassword
[root@server-0005 bin]# chown -R esroot:esgroup /data/elasticsearch
                  1
                  2
                  3

                  # 配置用户权限

                  编辑/etc/security/limits.conf,在最后面加上

                  esroot hard nofile 65536
esroot soft nofile 65536
                  1
                  2

                  编辑/etc/sysctl.conf，加上

                  vm.max_map_count=655360
                  1

                  执行命令

                  sysctl -p
                  1

                  # 运行es

                  # 切换用户
su esroot
# -d 是后台运行
bin/elasticsearch -d
